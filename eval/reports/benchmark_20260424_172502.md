  t2.cat_name, 
  SUM(t1.pay_amount) AS gmv
FROM 
  dwd_trade_order t1
JOIN 
  dim_sku_info t2 ON t1.sku_id = t2.sku_id
WHERE 
  t1.order_status IN (20, 30) AND 
  t2.is_on_sale = 1

  t2.cat_name

  gmv DESC
LIMIT 5;
